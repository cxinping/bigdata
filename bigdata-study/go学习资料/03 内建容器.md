# 数组



![](https://images.cnblogs.com/cnblogs_com/wangshuo1/1613306/o_220919123151_%E5%BE%AE%E4%BF%A1%E6%88%AA%E5%9B%BE_20220919191121.png)



例子

```
package main

import "fmt"

func main() {
	var arr1 [5]int
	arr2 := [3]int{1, 3, 5}
	arr3 := [...]int{2, 4, 6, 8, 10}
	var grid [4][5]int

	fmt.Println(arr1, arr2, arr3)
	fmt.Println(grid)

	for i, v := range arr3 {
		fmt.Println(i, v, arr3[i])
	}
}
```
